## Storage Rules (for image uploads) ##
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // ONLY allow users to read and write to the inventory_images folder
    match /inventory_images/{allPaths=**} {
      allow read, write: if request.auth != null;
    }
  }
}
